### Usage Tracker (Monitoring Only)

```python
# app/services/usage_tracker.py

class UsageTracker:
    """Track usage for monitoring - NO BLOCKING"""
    
    def __init__(self):
        self.redis = get_redis()
        self.targets = {
            'whatsapp_conversations': 5000,
            'voice_calls': 100,
            'llm_conversations': 3500,
        }
    
    async def track_whatsapp_conversation(self, msg_type: str):
        """Just count - no blocking"""
        month_key = datetime.now().strftime('%Y-%m')
        await self.redis.incr(f"usage:whatsapp:total:{month_key}")
        await self.redis.incr(f"usage:whatsapp:{msg_type}:{month_key}")
        await self._check_alert_threshold('whatsapp')
    
    async def track_voice_call(self, duration_seconds: int):
        """Track call duration"""
        month_key = datetime.now().strftime('%Y-%m')
        await self.redis.incr(f"usage:voice:calls:{month_key}")
        minutes = max(1, duration_seconds // 60)
        await self.redis.incrby(f"usage:voice:minutes:{month_key}", minutes)
        await self._check_alert_threshold('voice')
    
    async def track_llm_tokens(self, input_tokens: int, output_tokens: int, cached: bool):
        """Track LLM usage"""
        month_key = datetime.now().strftime('%Y-%m')
        key = f"usage:llm:input_{'cached' if cached else 'new'}:{month_key}"
        await self.redis.incrby(key, input_tokens)
        await self.redis.incrby(f"usage:llm:output:{month_key}", output_tokens)
        await self.redis.incr(f"usage:llm:conversations:{month_key}")
        await self._check_alert_threshold('llm')
    
    async def _check_alert_threshold(self, service: str):
        """Send alerts at 50%, 75%, 90% - but don't block"""
        usage = await self.get_current_usage()
        percentage = self._calculate_percentage(usage, service)
        
        if percentage >= 90:
            await self._send_alert(service, percentage, 'critical')
        elif percentage >= 75:
            await self._send_alert(service, percentage, 'warning')
        elif percentage >= 50:
            await self._send_alert(service, percentage, 'info')
```

# app/api/admin/usage.py

@router.get("/dashboard")
async def get_usage_dashboard():
    """Get current usage dashboard - READ ONLY"""
    tracker = UsageTracker()
    usage = await tracker.get_current_usage()
    costs = calculate_costs(usage)
    
    return {
        "usage": usage,
        "costs": costs,
        "budget": {
            "total": 15000,
            "api_budget": 12000,
            "used": costs['total'],
            "remaining": 12000 - costs['total'],
            "percentage": (costs['total'] / 12000) * 100
        },
        "status": "on_track" if costs['total'] < 12000 else "over_budget"
    }

def calculate_costs(usage):
    """Calculate current API costs"""
    return {
        'whatsapp': (usage['whatsapp']['utility'] * 0.12 + 
                    usage['whatsapp']['marketing'] * 0.85),
        'llm': ((usage['llm']['input_cached'] / 1_000_000) * 6 +
               (usage['llm']['input_new'] / 1_000_000) * 23 +
               (usage['llm']['output'] / 1_000_000) * 95),
        'voice': 1999 + (usage['voice']['minutes'] * 0.75),
        'fixed': 3779,  # Exotel + ElevenLabs + Bolna
        'total': # sum of above
    }
